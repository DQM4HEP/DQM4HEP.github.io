134 StatusCode DQMDataStream::reallocBuffer(dqm_uint newSize, bool copyCurrentBuffer)
135 {
136  dqm_uint bufferSize = getBufferSize();
137  dqm_uint currentPos = m_pBufferPtr - m_pBuffer;
138 
139  // can not shrink the current buffer or allocate 0 size buffer
140  if(newSize < bufferSize || 0 == newSize)
141  return STATUS_CODE_NOT_ALLOWED;
142 
143  char *pNewBuffer = new char[newSize];
144 
145  if(NULL != m_pBuffer)
146  {
147  if(copyCurrentBuffer && 0 != bufferSize)
148  {
149  memcpy(pNewBuffer, m_pBuffer, bufferSize);
150  }
151  else
152  {
153  memset(pNewBuffer, 0, newSize);
154  }

157  }
158  else
159  {

161  }
162

164  m_maxBufferSize = newSize;

167  return STATUS_CODE_SUCCESS;
