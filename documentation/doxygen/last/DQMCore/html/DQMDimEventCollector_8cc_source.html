379 void DQMDimEventCollector::updateEventService()
380 {
381  // if not running, do not update
382  if(!isRunning())
383  return;
384 
385  // event available ?
386  if(NULL == m_dataStream.getBuffer() || 0 == m_dataStream.getBufferSize())
387  return;
388 
389  int *clientIds = new int[m_clientMap.size()];
390  memset(&clientIds[0], 0, m_clientMap.size());
391  int currentId = 0;
392 
393  for(ClientMap::iterator iter = m_clientMap.begin(), endIter = m_clientMap.end() ;
394  endIter != iter ; ++iter)
395  {
396  // check for update mode
397  if(!iter->second.m_updateMode)
398  continue;
399 
400  // specific case where the client has queried a sub part of the event
401  if(!iter->second.m_subEventIdentifier.empty()
402  && (NULL != m_pEventStreamer && NULL != m_pCurrentEvent))
403  {
404  // try to serialize the sub event
405  m_subEventDataStream.reset();
406 
407  if(STATUS_CODE_SUCCESS != m_pEventStreamer->serialize(m_pCurrentEvent, iter->second.m_subEventIdentifier, &m_subEventDataStream))
408  continue;
409 
410  dqm_char *pEventBuffer = m_subEventDataStream.getBuffer();
411  int bufferSize = m_subEventDataStream.getBufferSize();
412 
413  if(NULL == pEventBuffer || 0 == bufferSize)
414  continue;
415 
416  int clientIdArray[2];
417  clientIdArray[0] = iter->first;
418  clientIdArray[1] = 0;
419 
420  m_pEventUpdateService->selectiveUpdateService((void *) pEventBuffer, bufferSize, &clientIdArray[0]);
421 
422  continue;
423  }
424 
425  clientIds[currentId] = iter->first;
426  currentId ++;
427  }
428 
429  if(currentId != 0)
430  m_pEventUpdateService->selectiveUpdateService((void *) m_dataStream.getBuffer(), m_dataStream.getBufferSize(), clientIds);
431 
432  delete [] clientIds;
433 
434  return;
435 }
