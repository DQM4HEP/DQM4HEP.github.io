272 void DQMRunControlService::handleCurrentRunRpc(DimRpc *pRpc)
273 {
274  if(!this->isRunning())
275  return;
276 
277  std::cout << "Handling current run rpc !" << std::endl;
278 
279  try
280  {
281  // if running send the current run
282  // else send back an empty run
283  if(m_pRunControl->isRunning())
284  {
285  // stream
286  m_dataStream.reset();
287  THROW_RESULT_IF(STATUS_CODE_SUCCESS, !=, m_pRunControl->getCurrentRun()->serialize(&m_dataStream));
288

291

294  }
295  else

303  // send back empty run
