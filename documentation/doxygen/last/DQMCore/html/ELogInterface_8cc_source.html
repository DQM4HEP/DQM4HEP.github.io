253 int ELog::processELogCommand() const
254 {
255  std::stringstream command;
256  command << m_eLogCommand
257  << " -h \"" << m_host << "\""
258  << " -p " << m_port
259  << " -u \"" << m_userName << "\" \"" << m_password << "\""
260  << " -l \"" << m_logbook << "\"";
261 
262  for(LogbookEntryMap::const_iterator iter = m_logbookEntryMap.begin(), endIter = m_logbookEntryMap.end() ;
263  endIter != iter ; ++iter)
264  {
265  command << " -a \"" << iter->first << "\"=\"" << iter->second << "\"";
266  }
267 
268  command << " -n " << m_encoding;
269 
270  if(m_useSSL)
271  command << " -s ";
272 
273  if(m_suppresssEmailNotification)
274  command << " -x ";
275 
276  command << " \"" << m_message << "\"";
277 
278  streamlog_out(MESSAGE) << "Processing logbook command : '" << command.str() << "'" << std::endl;
279  return system(command.str().c_str());
280 }
