143 inline StatusCode DQMDataStream::write(const T *pValue, dqm_uint nVals)
144 {
145  if(NULL == pValue)
146  return STATUS_CODE_INVALID_PARAMETER;
147 
148  dqm_uint size = sizeof(T)*nVals;
149  dqm_uint availableSize = getMaxBufferSize() - getBufferSize();
150 
151  if(availableSize <= size+sizeof(dqm_uint))
152  {
153  dqm_uint newSize = m_maxBufferSize + 1*1024*1024;
154 
155  if(size + sizeof(dqm_uint) > 1*1024*1024)
156  newSize = size + sizeof(dqm_uint);
157 
158  RETURN_RESULT_IF(STATUS_CODE_SUCCESS, !=, reallocBuffer(newSize));
159  }
160 
161  // get the current buffer ptr
162  dqm_uint *pCurrBuff = (dqm_uint *) m_pBufferPtr;
163 
164  // write the packet size
165  pCurrBuff[0] = size;
166 
167  // increment the buffer
168  m_pBufferPtr += sizeof(dqm_uint);
169  m_bufferSize += sizeof(dqm_uint);
170 
171  // write the packet in the buffer
172  memcpy(m_pBufferPtr, (const void*)pValue, size);
173 
174  // increment again the buffer
175  m_pBufferPtr += size;
176  m_bufferSize += size;
177 
178  return STATUS_CODE_SUCCESS;
