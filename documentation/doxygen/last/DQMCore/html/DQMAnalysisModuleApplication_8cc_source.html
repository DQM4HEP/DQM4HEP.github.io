371 StatusCode DQMAnalysisModuleApplication::configureNetwork(const TiXmlHandle xmlHandle)
372 {
373  try
374  {
375  streamlog_out(MESSAGE) << "DQMAnalysisModuleApplication::configureNetwork: configuring ..." << std::endl;
376 
377  //
378  // Get all xml attributes needed to configure network options !
379  //
380 
381  TiXmlElement *const pXmlElement = xmlHandle.FirstChildElement("network").Element();
382 
383  if(NULL == pXmlElement)
384  {
385  streamlog_out(ERROR) << "<network> element not found !" << std::endl;
386  return STATUS_CODE_NOT_FOUND;
387  }
388 
389  const TiXmlHandle networkHandle(pXmlElement);
390 
391  TiXmlElement *const pRunControlXmlElement = networkHandle.FirstChildElement("runcontrol").Element();
392  TiXmlElement *const pEventCollectorXmlElement = networkHandle.FirstChildElement("eventcollector").Element();
393  TiXmlElement *const pMonitorElementCollectorXmlElement = networkHandle.FirstChildElement("monitorelementcollector").Element();
394 
395  if(NULL == pRunControlXmlElement

397  || NULL == pMonitorElementCollectorXmlElement)

404  THROW_RESULT_IF_AND_IF(STATUS_CODE_SUCCESS, STATUS_CODE_NOT_FOUND, !=, DQMXmlHelper::getAttribute(pRunControlXmlElement, "type", m_settings.m_runControlType));
405  THROW_RESULT_IF_AND_IF(STATUS_CODE_SUCCESS, STATUS_CODE_NOT_FOUND, !=, DQMXmlHelper::getAttribute(pEventCollectorXmlElement, "type", m_settings.m_eventClientType));
406  THROW_RESULT_IF(STATUS_CODE_SUCCESS, !=, DQMXmlHelper::getAttribute(pMonitorElementCollectorXmlElement, "name", m_settings.m_monitorElementCollector));
407 
408  //
409  // Configure network form the extracted xml values
410  //
411 
412  // monitor element collector name
413  this->getMonitorElementSender()->setCollectorName(m_settings.m_monitorElementCollector);
414 
415  // event client
416  DQMEventClient *pEventClient = DQMPluginManager::instance()->createPluginClass<DQMEventClient>(m_settings.m_eventClientType);
417 
418  if( ! pEventClient )

428 
429  if(NULL == pRunControlClient)
430  throw StatusCodeException(STATUS_CODE_NOT_FOUND);
431 
432  pRunControlClient->setRunControlName( m_settings.m_runControlName );
433  m_pRunControlClient = pRunControlClient;
434 
435  streamlog_out(MESSAGE) << "DQMAnalysisModuleApplication::configureNetwork: configuring ... OK" << std::endl;
436  }
437  catch(const StatusCodeException &exception)
438  {
439  return exception.getStatusCode();
440  }
441 
442  return STATUS_CODE_SUCCESS;
