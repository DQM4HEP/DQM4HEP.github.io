245 StatusCode DQMMeanWithinExpectedTest::run(DQMMonitorElement *pMonitorElement)
246 {
247  m_isSuccessful = false;
248  m_quality = NO_QUALITY;
249  m_message = "";
250 
251  TH1 *pHistogram = pMonitorElement->get<TH1>();
252 
253  if(0 == m_strategy)
254  {
255  float mean = pHistogram->GetMean();
256 
257  // test failed ?
258  if(mean < m_xMin || mean > m_xMax)
259  {
260  m_quality = VERY_BAD_QUALITY;
261  m_message = "Out of range !";
262  }
263  else
264  {
265  m_quality = VERY_GOOD_QUALITY;
266  m_message = "";
267  }
268 
269  m_isSuccessful = true;
270  }
271  else if(1 == m_strategy)
272  {
273  float chi = (pHistogram->GetMean() - m_expectedMean)/m_sigma;
274  float probability = TMath::Prob(chi*chi, 1);
275 
276  m_quality = DQM4HEP::scaleToQuality(probability);
277  m_isSuccessful = true;
278  }
279  else if(2 == m_strategy)
280  {
281  float chi = (pHistogram->GetMean() - m_expectedMean)/pHistogram->GetRMS();
282  float probability = TMath::Prob(chi*chi, 1);
283 
284  m_quality = DQM4HEP::scaleToQuality(probability);
285  m_isSuccessful = true;
286  }
287  else
288  {
289  m_isSuccessful = false;
290  m_quality = NO_QUALITY;
291  m_message = "Undefined strategy for this test";
292  }
293 
294  return STATUS_CODE_SUCCESS;

371 StatusCode DQMChi2FitFunctionTest::run(DQMMonitorElement *pMonitorElement)
372 {
373  m_isSuccessful = false;
374  m_quality = NO_QUALITY;
375  m_message = "";
376 
377  TH1 *pHistogram = pMonitorElement->get<TH1>();
378 
379  std::string functionName = "DQMChi2FitFunctionTest_"+ getName() +"_fitFunction";
380 
381  // we need to remove the function from the TH1
382  // since when TH1::Fit is called, the function
383  // is stored in the TH1. The function is removed and deleted
384  // since we don't know if the formula is the same as the user provides
385  TF1 *pFitFunction = pHistogram->GetFunction(functionName.c_str());
386 
387  if(pFitFunction)
388  {
389  pHistogram->RecursiveRemove(pFitFunction);
390  delete pFitFunction;
391  pFitFunction = NULL;
392  }
393 
394  // declare the fit function and fill the parameter list
395  pFitFunction = new TF1(functionName.c_str(), m_formula.c_str(), m_functionXMin, m_functionXMax);//, m_functionNParameters);
396 
397  // set the parameters depending on if an
398  // input has been provided or not by the user
399  for(unsigned int p=0 ; p<m_functionNParameters ; p++)
400  pFitFunction->SetParameter(p, m_inputParameterList.empty() ? 0 : m_inputParameterList.at(p));
401 
402  // Q for quiet
403  // N for no draw
404  int status = pHistogram->Fit(pFitFunction, "QN", "", m_functionXMin, m_functionXMax);
405 
406  if(status != 0)
407  {
408  m_message = "Try to fit function '"+ m_formula + "' with histogram '" + pMonitorElement->getName() + "'. " +
409  "Fit failed and returned status " + DQM4HEP::typeToString(status);
410 
411  return STATUS_CODE_SUCCESS;
412  }
413 
414  double chi2 = pFitFunction->GetChisquare();
415  double proba = TMath::Prob(chi2, m_inputParameterList.size());
416 
417  m_isSuccessful = true;
418  m_quality = DQM4HEP::scaleToQuality(proba);
419  m_message = "Function '"+ m_formula +"' fitted on histogram '" + pMonitorElement->getName() + "' ends with the following output :\n" +
420  "Chi2 = " + DQM4HEP::typeToString(chi2) + "\n" +
421  "Excess probability = " + DQM4HEP::typeToString(proba) + "\n" +
422  "Output parameters : \n";
423 
424  for(unsigned int p=0 ; p<m_functionNParameters ; p++)
425  m_message += (" p" + DQM4HEP::typeToString(p) + " = " + DQM4HEP::typeToString(pFitFunction->GetParameter(p)) + "\n");
426 
427  return STATUS_CODE_SUCCESS;
428 }
