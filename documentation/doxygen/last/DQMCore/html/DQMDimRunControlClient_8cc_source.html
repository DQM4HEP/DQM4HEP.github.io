130  if(m_pStartOfRunInfo == pCurrentDimInfo)

136  dqm_uint bufferSize = pCurrentDimInfo->getSize();
137 
138  if(pBuffer == NULL || bufferSize == 0)
139  return;
140 
141  m_dataStream.reset();
142  THROW_RESULT_IF(STATUS_CODE_SUCCESS, !=, m_dataStream.setBuffer(pBuffer, bufferSize));
143 
144  DQMRun *pRun = new DQMRun();
145 
146  try

154  THROW_RESULT_IF(STATUS_CODE_SUCCESS, !=, this->startNewRun(pRun));

161  }
