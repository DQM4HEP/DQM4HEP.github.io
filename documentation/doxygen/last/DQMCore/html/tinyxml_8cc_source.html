55 void TiXmlBase::EncodeString( const TIXML_STRING& str, TIXML_STRING* outString )
56 {
57  int i=0;
58 
59  while( i<(int)str.length() )
60  {
61  unsigned char c = (unsigned char) str[i];
62 
63  if ( c == '&'
64  && i < ( (int)str.length() - 2 )
65  && str[i+1] == '#'
66  && str[i+2] == 'x' )
67  {
68  // Hexadecimal character reference.
69  // Pass through unchanged.
70  // &#xA9; -- copyright symbol, for example.
71  //
72  // The -1 is a bug fix from Rob Laveaux. It keeps
73  // an overflow from happening if there is no ';'.
74  // There are actually 2 ways to exit this loop -
75  // while fails (error case) and break (semicolon found).
76  // However, there is no mechanism (currently) for
77  // this function to return an error.
78  while ( i<(int)str.length()-1 )
79  {
80  outString->append( str.c_str() + i, 1 );
81  ++i;
82  if ( str[i] == ';' )
83  break;
84  }
85  }
86  else if ( c == '&' )
87  {
88  outString->append( entity[0].str, entity[0].strLength );
89  ++i;
90  }
91  else if ( c == '<' )
92  {
93  outString->append( entity[1].str, entity[1].strLength );
94  ++i;
95  }
96  else if ( c == '>' )
97  {
98  outString->append( entity[2].str, entity[2].strLength );
99  ++i;
100  }
101  else if ( c == '\"' )
102  {
103  outString->append( entity[3].str, entity[3].strLength );
104  ++i;
105  }
106  else if ( c == '\'' )
107  {
108  outString->append( entity[4].str, entity[4].strLength );
109  ++i;
110  }
111  else if ( c < 32 )
112  {
113  // Easy pass at non-alpha/numeric/symbol
114  // Below 32 is symbolic.
115  char buf[ 32 ];
116 
117  #if defined(TIXML_SNPRINTF)
118  TIXML_SNPRINTF( buf, sizeof(buf), "&#x%02X;", (unsigned) ( c & 0xff ) );
119  #else
120  sprintf( buf, "&#x%02X;", (unsigned) ( c & 0xff ) );
121  #endif
122 
123  //*ME: warning C4267: convert 'size_t' to 'int'
124  //*ME: Int-Cast to make compiler happy ...
125  outString->append( buf, (int)strlen( buf ) );
126  ++i;
127  }
128  else
129  {
130  //char realc = (char) c;
131  //outString->append( &realc, 1 );
132  *outString += (char) c; // somewhat more efficient function call.
133  ++i;
134  }
135  }
136 }

760 {
761  int i;
762  assert( cfile );
763  for ( i=0; i<depth; i++ ) {
764  fprintf( cfile, " " );
765  }
766 
767  fprintf( cfile, "<%s", value.c_str() );
768 
769  const TiXmlAttribute* attrib;

771  {

774  }
775

784  }

796  {

798  {
799  fprintf( cfile, "\n" );
800  }

944 bool TiXmlDocument::LoadFile( FILE* file, TiXmlEncoding encoding )
945 {
946  if ( !file )
947  {
948  SetError( TIXML_ERROR_OPENING_FILE, 0, 0, TIXML_ENCODING_UNKNOWN );
949  return false;
950  }
951 
952  // Delete the existing data:
953  Clear();
954  location.Clear();
955 
956  // Get the file size, so we can pre-allocate the string. HUGE speed impact.
957  long length = 0;
958  fseek( file, 0, SEEK_END );
959  length = ftell( file );
960  fseek( file, 0, SEEK_SET );
961 
962  // Strange case, but good to handle up front.
963  if ( length <= 0 )
964  {
965  SetError( TIXML_ERROR_DOCUMENT_EMPTY, 0, 0, TIXML_ENCODING_UNKNOWN );
966  return false;
967  }
968 
969  // Subtle bug here. TinyXml did use fgets. But from the XML spec:
970  // 2.11 End-of-Line Handling
971  // <snip>
972  // <quote>
973  // ...the XML processor MUST behave as if it normalized all line breaks in external
974  // parsed entities (including the document entity) on input, before parsing, by translating
975  // both the two-character sequence #xD #xA and any #xD that is not followed by #xA to
976  // a single #xA character.
977  // </quote>
978  //
979  // It is not clear fgets does that, and certainly isn't clear it works cross platform.
980  // Generally, you expect fgets to translate from the convention of the OS to the c/unix
981  // convention, and not work generally.
982 
983  /*
984  while( fgets( buf, sizeof(buf), file ) )
985  {
986  data += buf;
987  }
988  */
989 
990  char* buf = new char[ length+1 ];
991  buf[0] = 0;
992 
993  if ( fread( buf, length, 1, file ) != 1 ) {
994  delete [] buf;
995  SetError( TIXML_ERROR_OPENING_FILE, 0, 0, TIXML_ENCODING_UNKNOWN );
996  return false;
997  }
998 
999  // Process the buffer in place to normalize new lines. (See comment above.)
1000  // Copies from the 'p' to 'q' pointer, where p can advance faster if
1001  // a newline-carriage return is hit.
1002  //
1003  // Wikipedia:
1004  // Systems based on ASCII or a compatible character set use either LF (Line feed, '\n', 0x0A, 10 in decimal) or
1005  // CR (Carriage return, '\r', 0x0D, 13 in decimal) individually, or CR followed by LF (CR+LF, 0x0D 0x0A)...
1006  // * LF: Multics, Unix and Unix-like systems (GNU/Linux, AIX, Xenix, Mac OS X, FreeBSD, etc.), BeOS, Amiga, RISC OS, and others
1007  // * CR+LF: DEC RT-11 and most other early non-Unix, non-IBM OSes, CP/M, MP/M, DOS, OS/2, Microsoft Windows, Symbian OS
1008  // * CR: Commodore 8-bit machines, Apple II family, Mac OS up to version 9 and OS-9
1009 
1010  const char* p = buf; // the read head
1011  char* q = buf; // the write head
1012  const char CR = 0x0d;
1013  const char LF = 0x0a;
1014 
1015  buf[length] = 0;
1016  while( *p ) {
1017  assert( p < (buf+length) );
1018  assert( q <= (buf+length) );
1019  assert( q <= p );
1020 
1021  if ( *p == CR ) {
1022  *q++ = LF;
1023  p++;
1024  if ( *p == LF ) { // check for CR+LF (and skip LF)
1025  p++;
1026  }
1027  }
1028  else {
1029  *q++ = *p++;
1030  }
1031  }
1032  assert( q <= (buf+length) );
1033  *q = 0;
1034 
1035  Parse( buf, 0, encoding );
1036 
1037  delete [] buf;
1038  return !Error();
1039 }
