625 void DQMMonitorElementCollector::handleMeCollectUpdate(DimCommand *pCommand)
626 {
627  try
628  {
629  char *pBuffer = static_cast<char *>(pCommand->getData());
630  int bufferSize = pCommand->getSize();
631 
632  if(bufferSize == 0 || NULL == pBuffer)
633  {
634  streamlog_out(WARNING) << "Empty buffer, skipping packet ..." << std::endl;
635  throw StatusCodeException(STATUS_CODE_SUCCESS);
636  }
637 
638  m_dataStream.reset();
639  THROW_RESULT_IF(STATUS_CODE_SUCCESS, !=, m_dataStream.setBuffer(pBuffer, bufferSize));
640 
641  DQMMonitorElementPublication publication;
642  THROW_RESULT_IF(STATUS_CODE_SUCCESS, !=, publication.deserialize(&m_dataStream));
643 
644  if(publication.empty())
645  {
646  streamlog_out(WARNING) << "Empty publication, skipping packet ..." << std::endl;
647  throw StatusCodeException(STATUS_CODE_SUCCESS);
648  }
649 
650  int nElements = 0;
651  ClientUpdateMap clientUpdateMap;
652 
653  // update the monitor element with ones received !
654  // build the monitor element list to update for each client
655  for(DQMMonitorElementPublication::iterator iter = publication.begin(),
656  endIter = publication.end() ; endIter != iter ; ++iter)
657  {
658  ModuleMeInfoMap::iterator findIter = m_moduleMeInfoMap.find(iter->first);
659 
660  // if module not found, clean the monitor elements related to this module
661  if(findIter == m_moduleMeInfoMap.end())
662  {
663  for(DQMMonitorElementList::iterator meIter = iter->second.begin(), meEndIter = iter->second.end() ;
664  meEndIter != meIter ; ++meIter)
665  delete *meIter;
666 
667  continue;
668  }
669 
670  // update monitor elements !
671  for(DQMMonitorElementList::iterator meIter = iter->second.begin(), meEndIter = iter->second.end() ;
672  meEndIter != meIter ; ++meIter)
673  {
674  bool updated = findIter->second->updateMonitorElement(*meIter);
675 
676  if(!updated)
677  delete *meIter;
678 
679  nElements++;
680 
681  (*meIter)->setCollectorName(this->getCollectorName());
682 
683  // get the client ids that have subscribed to these elements
684  std::string meFullName = ( (*meIter)->getPath() + (*meIter)->getName() ).getPath();
685 
686  if(meFullName.at(0) != '/')
687  meFullName = "/" + meFullName;
688 
689  streamlog_out(DEBUG) << "meFullName on collect : " << meFullName << std::endl;
690  std::set<int> clientIdSet = findIter->second->getSubscribers(meFullName);
691 
692  // add the monitor element entry into the update map
693  for(std::set<int>::iterator cliIter = clientIdSet.begin(), cliEndIter = clientIdSet.end() ;
694  cliEndIter != cliIter ; ++cliIter)
695  {
696  clientUpdateMap[*cliIter].insert(*meIter);
697  }
698  }
699  }
700 
701  m_pStatisticsService->update(nElements);
702 
703  streamlog_out(DEBUG) << "Received n elements : " << nElements << " from " << publication.size() << " modules" << std::endl;
704  this->updateClients(clientUpdateMap);
705  }
706  catch(StatusCodeException &exception)
707  {
708  streamlog_out(WARNING) << "Exception caught : " << exception.toString() << std::endl;
709  }
710 }
