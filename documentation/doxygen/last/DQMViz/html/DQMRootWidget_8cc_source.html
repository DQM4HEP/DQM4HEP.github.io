231  pContextMenu->addAction("Remove", this, SLOT(removeCanvas()));

239  pSetDrawOptionMenu->setEnabled(false);

431 void DQMRootWidget::updateMonitorElement(DQMGuiMonitorElement *pNewGuiMonitorElement)
432 {
433  DQMGuiMonitorElement *pCurrentGuiMonitorElement = this->getCurrentMonitorElement();
434 
435  if(pCurrentGuiMonitorElement == pNewGuiMonitorElement)
436  return;
437 
438  if(NULL != pCurrentGuiMonitorElement)
439  {
440  QObject::disconnect(pCurrentGuiMonitorElement, SIGNAL(destroyed()), this, SLOT(removeCanvas()));
441  QObject::disconnect(pCurrentGuiMonitorElement, SIGNAL(updated()), this, SLOT(redraw()));
442  }
443 
444  m_pCurrentMonitorElement = pNewGuiMonitorElement;
445 
446  if(pNewGuiMonitorElement)
447  {
448  QObject::connect(pNewGuiMonitorElement, SIGNAL(destroyed()), this, SLOT(removeCanvas()));
449  QObject::connect(pNewGuiMonitorElement, SIGNAL(updated()), this, SLOT(redraw()));
450 
451  DQMCanvas *pCanvas = this->getCanvas();
452 
453  pCanvas->setWindowTitle(pNewGuiMonitorElement->getMonitorElement()->getTitle().c_str());
454  pCanvas->setWindowIcon(pCanvas->getMonitoring()->getController()->getIcon(pNewGuiMonitorElement->getMonitorElement()->getQuality()));
455  }
