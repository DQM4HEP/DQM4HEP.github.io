558 void DQMJobInterfaceWidget::loadJson(const Json::Value &root)
559 {
560  m_pTreeWidget->clear();
561 
562  StringVector hostList = root["HOSTS"].getMemberNames();
563 
564  // loop over hosts
565  for(StringVector::iterator iter = hostList.begin(), endIter = hostList.end() ;
566  endIter != iter ; ++iter)
567  {
568  const Json::Value &host(root["HOSTS"][*iter]);
569 
570  QStringList name((*iter).c_str());
571 
572  QTreeWidgetItem *pHostItem = new QTreeWidgetItem(name);
573 
574  pHostItem->setData(0, Qt::UserRole, HOST_ITEM);
575  m_pTreeWidget->addTopLevelItem(pHostItem);
576 
577  // loop over jobs host
578  for(unsigned int j=0 ; j<host.size() ; j++)
579  {
580  QStringList jobHeader;
581  jobHeader << host[j]["NAME"].asString().c_str() << host[j]["PROGRAM"].asString().c_str() << host[j]["PID"].asString().c_str() << host[j]["STATUS"].asString().c_str() ;
582 
583  QTreeWidgetItem *pJobItem = new QTreeWidgetItem(jobHeader);
584 
585  pJobItem->setData(NAME, Qt::UserRole, JOB_ITEM);
586  pJobItem->setData(PROG_NAME, Qt::UserRole, JOB_PROGRAM_NAME_ITEM);
587  pJobItem->setData(PID, Qt::UserRole, JOB_PID_ITEM);
588  pJobItem->setData(STATUS, Qt::UserRole, JOB_STATUS_ITEM);
589 
590  pHostItem->addChild(pJobItem);
591 
592  QTreeWidgetItem *pEnvironmentItem = new QTreeWidgetItem(QStringList("ENV"));
593  pEnvironmentItem->setData(0, Qt::UserRole, JOB_ENV_ITEM);
594  pJobItem->addChild(pEnvironmentItem);
595 
596  const Json::Value &envJsonValue = host[j]["ENV"];
597 
598  for(unsigned int e=0 ; e<envJsonValue.size() ; e++)
599  {
600  QString env(envJsonValue[e].asString().c_str());
601 
602  QStringList envVarHeader;
603  envVarHeader << env;
604 
605  QTreeWidgetItem *pEnvVarItem = new QTreeWidgetItem(envVarHeader);
606  pEnvVarItem->setData(0, Qt::UserRole, VARIABLE_ITEM);
607  pEnvironmentItem->addChild(pEnvVarItem);
608  }
609 
610  QTreeWidgetItem *pArgsItem = new QTreeWidgetItem(QStringList("ARGS"));
611  pArgsItem->setData(0, Qt::UserRole, JOB_ARGS_ITEM);
612  pJobItem->addChild(pArgsItem);
613 
614  const Json::Value &argsJsonValue = host[j]["ARGS"];
615 
616  for(unsigned int e=0 ; e<argsJsonValue.size() ; e++)
617  {
618  QString arg(argsJsonValue[e].asString().c_str());
619 
620  QStringList argVarHeader;
621  argVarHeader << arg;
622 
623  QTreeWidgetItem *pArgVarItem = new QTreeWidgetItem(argVarHeader);
624  pArgVarItem->setData(0, Qt::UserRole, VARIABLE_ITEM);
625  pArgsItem->addChild(pArgVarItem);
626  }
627  }
628 
629  pHostItem->setExpanded(true);
630  }
631 
632  m_pTreeWidget->header()->resizeSections(QHeaderView::Stretch);
633 }

637 void DQMJobInterfaceWidget::updateStatus(const Json::Value &value)
638 {
639  QMap<QString, QColor> stateToColorMap;
640  stateToColorMap["Z"] = QColor(Qt::red); // zombie
641  stateToColorMap["T"] = QColor(Qt::darkRed); // traced or stopped
642  stateToColorMap["R"] = QColor(Qt::green); // running
643  stateToColorMap["D"] = QColor(Qt::darkGreen); // un-interruptible sleep
644  stateToColorMap["S"] = QColor(Qt::darkGreen); // interruptible sleep
645  stateToColorMap["X"] = QColor(Qt::red); // interruptible sleep
646 
647  for(unsigned int i=0 ; i<m_pTreeWidget->topLevelItemCount() ; i++)
648  {
649  QTreeWidgetItem *pHostItem = m_pTreeWidget->topLevelItem(i);
650 
651  std::string hostName = pHostItem->text(NAME).toStdString();
652 
653  for(unsigned int j=0 ; j<pHostItem->childCount() ; j++)
654  {
655  QTreeWidgetItem *pJobItem = pHostItem->child(j);
656 
657  std::string jobName = pJobItem->text(NAME).toStdString();
658 
659  bool found = false;
660 
661  for(unsigned int v=0 ; v<value.size() ; v++)
662  {
663  std::string vHost = value[v]["HOST"].asString();
664  std::string vJobName = value[v]["NAME"].asString();
665  uint32_t vJobPid = value[v]["PID"].asUInt();
666  std::string vJobStatus = value[v]["STATUS"].asString();
667 
668  if(vHost == hostName && vJobName == jobName)
669  {
670  pJobItem->setText(STATUS, QString(vJobStatus.c_str()).trimmed());
671  pJobItem->setText(PID, QString::number(vJobPid));
672 
673  found = true;
674  break;
675  }
676  }
677 
678  if(!found)
679  {
680  pJobItem->setText(STATUS, "X (dead)");
681  pJobItem->setText(PID, "");
682  }

728 void DQMJobInterfaceWidget::updateStatus(const QString &hostName)
729 {
730  const Json::Value value = m_pJobIterface->processStatus(hostName.toStdString())["JOBS"];
731  QTreeWidgetItem *pHostItem = 0;
732 
733  for(unsigned int i=0 ; i<m_pTreeWidget->topLevelItemCount() ; i++)
734  {
735  QTreeWidgetItem *pItem = m_pTreeWidget->topLevelItem(i);
736 
737  if(hostName == pItem->text(NAME))
738  pHostItem = pItem;
739  }
740 
741  if(!pHostItem)
742  return;
743 
744  QMap<QString, QColor> stateToColorMap;
745  stateToColorMap["Z"] = QColor(Qt::red); // zombie
746  stateToColorMap["T"] = QColor(Qt::darkRed); // traced or stopped
747  stateToColorMap["R"] = QColor(Qt::green); // running
748  stateToColorMap["D"] = QColor(Qt::darkGreen); // un-interruptible sleep
749  stateToColorMap["S"] = QColor(Qt::darkGreen); // interruptible sleep
750  stateToColorMap["X"] = QColor(Qt::red); // interruptible sleep
751 
752  for(unsigned int j=0 ; j<pHostItem->childCount() ; j++)
753  {
754  QTreeWidgetItem *pJobItem = pHostItem->child(j);
755 
756  std::string jobName = pJobItem->text(NAME).toStdString();
757 
758  bool found = false;
759 
760  for(unsigned int v=0 ; v<value.size() ; v++)
761  {
762  std::string vHost = value[v]["HOST"].asString();
763  std::string vJobName = value[v]["NAME"].asString();
764  uint32_t vJobPid = value[v]["PID"].asUInt();
765  std::string vJobStatus = value[v]["STATUS"].asString();
766 
767  if(vJobName == jobName)
768  {
769  pJobItem->setText(STATUS, QString(vJobStatus.c_str()).trimmed());
770  pJobItem->setText(PID, QString::number(vJobPid));
771 
772  found = true;
773  break;
774  }
775  }
776 
777  if(!found)
778  {
779  pJobItem->setText(STATUS, "X (dead)");
780  pJobItem->setText(PID, "");
781  }
782 
783  // cases available in /proc/pid/status file
784  QMap<QString, QColor>::iterator findIter = stateToColorMap.find(pJobItem->text(STATUS).at(0));
785 
786  if(findIter != stateToColorMap.end())
787  pJobItem->setData(STATUS, Qt::ForegroundRole, QBrush(findIter.value()));
788  else
789  pJobItem->setData(STATUS, Qt::ForegroundRole, QBrush(Qt::black));
790  }
791 
792  m_pTreeWidget->header()->resizeSections(QHeaderView::Stretch);
793 }
