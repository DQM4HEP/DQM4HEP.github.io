120 
121  if(m_clientMap.end() != findIter)
122  return findIter->second;

124  DQMGuiMonitorElementClient *pNewClient = this->createClient(collectorName);
125

136  m_clientMap.insert(DQMGuiMonitorElementClientMap::value_type(collectorName, pNewClient));

364 void DQMMonitoringController::handleServerStartup()
365 {
366  DQMGuiMonitorElementClient *pGuiClient = qobject_cast<DQMGuiMonitorElementClient *>(sender());
367 
368  if(!pGuiClient)
369  return;
370 
371  DQMMonitorElementClient *pClient = pGuiClient->getMonitorElementClient();
372  std::string collectorName = pClient->getCollectorName();
373 
374  std::cout << collectorName << " server startup !" << std::endl;
375 
376  // get all subscribed elements
377  DQMMonitorElementView *pMeView = this->getMonitoring()->getView()->getMonitorElementView();
378  pMeView->enableSubscription(collectorName, true);
379  QList<QTreeWidgetItem *> items = pMeView->getCheckedMonitorElements(collectorName);
380 
381  DQMMonitorElementRequest request;
382 
383  for(int j=0 ; j<items.size() ; j++)
384  {
385  DQMMonitorElementNavigator *pNavigator = qobject_cast<DQMMonitorElementNavigator *>(items.at(j)->treeWidget());
386 
387  std::string moduleName = pNavigator->getModuleName(items.at(j)).toStdString();
388  std::string path = pNavigator->getFullPathName(items.at(j)).toStdString();
389  std::string name = items.at(j)->text(0).toStdString();
390  std::string fullName = (DQMPath(path) + name).getPath();
391 
392  request.push_back(DQMMonitorElementRequest::value_type(moduleName, fullName));
393  }
394 
395  if(!pClient->isConnectedToService())
396  pClient->connectToService();
397 
398  pClient->replaceSubscription(request);
399 }

486 void DQMMonitoringController::openFile(const std::string &fileName)
487 {
488  TiXmlDocument document(fileName);
489 
490  if(!document.LoadFile())
491  {
492  QMessageBox::warning(this->getMonitoring()->getView()->getMainWindow(),
493  "XML Import",
494  "Couldn't import selection from xml file :\n\n"
495  "* Invalid XML file content");
496 
497  return;
498  }
499 
500  TiXmlHandle documentHandle(&document);
501  TiXmlElement *pRootElement = documentHandle.FirstChildElement().Element();
502 
503  if(!pRootElement)
504  {
505  QMessageBox::warning(this->getMonitoring()->getView()->getMainWindow(),
506  "XML Import",
507  "Couldn't import selection from xml file :\n\n"
508  "* No root element found");
509 
510  return;
511  }
512 
513  TiXmlHandle rootHandle(pRootElement);
514 
515  TiXmlElement *pMonitoringElement = rootHandle.FirstChild("monitoring").Element();
516 
517  if(!pMonitoringElement)
518  {
519  QMessageBox::warning(this->getMonitoring()->getView()->getMainWindow(),
520  "XML Import",
521  "Couldn't import selection from xml file :\n\n"
522  "* No <monitoring> element found");
523 
524  return;
525  }
526 
527  TiXmlHandle monitoringHandle(pMonitoringElement);
528 
529  this->clear();
530  this->getMonitoring()->getModel()->clear();
531  this->getMonitoring()->getView()->clear();
532 
533  TiXmlElement *pDimDnsNodeElement = monitoringHandle.FirstChild("dimDnsNode").Element();
534 
535  if(pDimDnsNodeElement)
536  {
537  std::string dimDnsNode;
538 
539  if(!pDimDnsNodeElement->QueryStringAttribute("value", &dimDnsNode))
540  DimClient::setDnsNode(dimDnsNode.c_str());
541  }
542 
543  TiXmlElement *pViewElement = monitoringHandle.FirstChild("view").Element();
544 
545  if(pViewElement)
546  this->getMonitoring()->getView()->fromXml(pViewElement);
547 
548  TiXmlElement *pControllerElement = monitoringHandle.FirstChild("controller").Element();
549 
550  if(pControllerElement)
551  {
552  TiXmlHandle controllerHandle(pControllerElement);
553 
554  for (TiXmlElement *pCollectorElement = controllerHandle.FirstChild("collector").Element(); NULL != pCollectorElement;
555  pCollectorElement = pCollectorElement->NextSiblingElement("collector"))
556  {
557  std::string collectorName;
558  if(pCollectorElement->QueryStringAttribute("name", &collectorName))
559  continue;
560 
561  this->getClient(collectorName);
562  }
563  }
564 }

594

603 
604  if(pViewElement)
605  pMonitoringElement->LinkEndChild(pViewElement);
606 
607  TiXmlElement *pControllerElement = new TiXmlElement("controller");
608 
609  for(DQMGuiMonitorElementClientMap::const_iterator iter = m_clientMap.begin(), endIter = m_clientMap.end() ;
610  endIter != iter ; ++iter)
611  {
612  const std::string &collectorName = iter->first;
613 
614  TiXmlElement *pCollectorElement = new TiXmlElement("collector");
615  pControllerElement->LinkEndChild(pCollectorElement);
616  pCollectorElement->SetAttribute("value", collectorName);
617  }
618 
619  pMonitoringElement->LinkEndChild(pControllerElement);
620

769 void DQMMonitoringController::openInROOTWindow(DQMGuiMonitorElement *pMonitorElement) const
770 {
771  TObject *pObject = pMonitorElement->getMonitorElement()->getObject();
772 
773  if(!pObject)
774  return;
775 
776  pid_t pid = fork();
777 
778  // parent process
779  if(pid != 0)
780  return;
781 
782  // child process
783  char templateFile[] = "/tmp/canvasTmpXXXXXX.root";
784  int ret = mkstemps(templateFile, 5);
785 
786  if(ret < 0)
787  exit(-1);
788 
789  std::string tmpRootFileName = templateFile;
790  TFile *pTmpRootFile = TFile::Open(tmpRootFileName.c_str(), "RECREATE");
791 
792  if(!pTmpRootFile)
793  exit(1);
794 
795  // re-add the pave stats
796  TH1 *pHistogram = (TH1*)pObject;
797 
798  if(pHistogram)
799  pHistogram->SetStats(1);
800 
801  pTmpRootFile->cd();
802  pObject->Write("dqm4hep_open_root_canvas");
803  delete pTmpRootFile;
804 
805  // full program name
806  std::string fullPathProg = std::string(DQMCore_DIR) + "/bin/dqm4hep_open_root_canvas";
807  char executivePath[1000];
808  memset(executivePath, 0, 1000);
809 
810  memcpy(executivePath, fullPathProg.c_str(), fullPathProg.size());
811 
812  // fill arguments
813  const unsigned int maxNArgs = 100;
814  const unsigned int maxMArgs = 1000;
815 
816  char argv[maxNArgs][maxMArgs];
817  char *pArgv[maxNArgs];
818 
819  for(unsigned int i=0 ; i<maxNArgs ; i++)
820  for(unsigned int j=0 ; j<maxMArgs ; j++)
821  argv[i][j] = char(NULL);
822 
823  pArgv[0] = executivePath;
824  sprintf(argv[1], "%s", "-f");
825  pArgv[1] = & argv[1][0];
826  sprintf(argv[2], "%s", tmpRootFileName.c_str());
827  pArgv[2] = & argv[2][0];
828  pArgv[3] = NULL;
829 
830  // fill environment
831  const unsigned int maxNEnv = 10;
832  const unsigned int maxMEnv = 1000;
833 
834  char env[maxNEnv][maxMEnv];
835  char *pEnv[maxNEnv];
836 
837  for(unsigned int i=0 ; i<maxNEnv ; i++)
838  for(unsigned int j=0 ; j<maxMEnv ; j++)
839  env[i][j] = char(NULL);
840 
841  sprintf(env[0], "%s", "DISPLAY=:0");
842  pEnv[0] = & env[0][0];
843  pEnv[1] = NULL;
844 
845  // start root panel from dqm4hep specific executable
846  int execveRet = execve(executivePath, pArgv, pEnv);
847 
848  streamlog_out(ERROR) << "ERROR ! execve returned " << execveRet << " with errno set to " << errno << std::endl;
849  streamlog_out(ERROR) << "Aborting" << std::endl;
850 
851  exit(1);
852 }

856 void DQMMonitoringController::saveAs(DQMCanvasArea *pCanvasArea)
857 {

859 
860  if(subWindows.isEmpty())

862 
863  QStringList formatList;
864  QStringList extensionList;
865  QString selectedFormat;
866 
867  formatList
868 // << "Postscript (*.ps)"
869 // << "Encapsulated Postscript (*.eps)"
870 // << "PDF (*.pdf)"
871 // << "GIF (*.gif)"
872 // << "JPG (*.jpg)"
873 // << "PNG (*.png)"
874 // << "ROOT Macro (*.C)"
875  << "ROOT Files (*.root)";
876 
877  extensionList
878 // << ".ps"
879 // << ".eps"
880 // << ".pdf"
881 // << ".gif"
882 // << ".jpg"
883 // << ".png"
884 // << ".C"
885  << ".root";
886 
887  QString fileName = QFileDialog::getSaveFileName(this->getMonitoring()->getView()->getMainWindow(),
888  "Save as",
889  "",
890  formatList.join(";;"),
891  &selectedFormat);
892 
893  if(fileName.isEmpty())
894  return;
895 
896  int lastIndexOf = fileName.lastIndexOf(extensionList.at(formatList.indexOf(selectedFormat)));
897  QString baseFileName;
898  QString extensionFileName;
899 
900  if(lastIndexOf < 0)
901  {
902  lastIndexOf = fileName.size() -1;
903  baseFileName = fileName;
904  int index = formatList.indexOf(selectedFormat);
905  extensionFileName = extensionList.at(index);
906  }
907  else
908  {
909  baseFileName = fileName.left(lastIndexOf);
910  extensionFileName = fileName.right(fileName.size() - lastIndexOf);
911  }
912 
913  std::string realFileName = (baseFileName + extensionFileName).toStdString();
914 
915  TFile *pFile = TFile::Open(realFileName.c_str(), "RECREATE");
916 
917  if(NULL == pFile)
918  return;
919 
920  for(int i=0 ; i<subWindows.count() ; i++)
921  {
922  DQMCanvas *pCanvas = qobject_cast<DQMCanvas*>(subWindows.at(i));
923 
924  if(!pCanvas)
925  continue;
926 
927  TCanvas *pSaveCanvas = pCanvas->getRootWidget()->GetCanvas();
928  std::string objectName = pCanvas->getCurrentMonitorElement()->getMonitorElement()->getName();
929  pFile->WriteTObject(pSaveCanvas, objectName.c_str());
930  }
931 
932  pFile->Close();
933  delete pFile;
