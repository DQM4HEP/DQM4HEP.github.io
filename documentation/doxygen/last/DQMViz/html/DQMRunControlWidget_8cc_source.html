291 void DQMRunControlWidget::startNewRun()
292 {
293  if(RUNNING_STATE == m_pRunControlService->getRunState())
294  {
295  try
296  {
297  if(m_pLogger)
298  m_pLogger->log(WARNING, "Stopping current run");
299

301  }
302  catch(StatusCodeException &exception)
303  {
304  if(m_pLogger)
305  m_pLogger->log(ERROR, "Couldn't end current run : " + exception.toString());
306  return;
307  }
308  }
309 
310  try
311  {
312  int runNumber = m_pRunNumberSpinBox->value();
313 
314  if(m_pLogger)
315  {
316  std::stringstream message;
317  message << "Starting new run no " << runNumber;
318  m_pLogger->log(MESSAGE, message.str());
319  }
320 
321  DQMRun *pRun = new DQMRun(runNumber, getDescription().toStdString(), getDetectorName().toStdString());
322 
323  for(unsigned int i=0 ; i<m_pRunParametersTableWidget->rowCount() ; i++)
324  {
325  QTableWidgetItem *pKeyItem = m_pRunParametersTableWidget->item(i, 0);
326  QTableWidgetItem *pValueItem = m_pRunParametersTableWidget->item(i, 1);
327 
328  if(!pKeyItem || !pValueItem)
329  continue;
330 
331  pRun->setParameter(pKeyItem->text().toStdString(), pValueItem->text().toStdString());
332  }
333 
334  THROW_RESULT_IF(STATUS_CODE_SUCCESS, !=, m_pRunControlService->startNewRun(pRun));
335 
336  time_t startTime = m_pRunControlService->getCurrentRun()->getStartTime();
337  std::string startTimeStdStr;
338  DQMCoreTool::timeToHMS(startTime, startTimeStdStr);
339 
340  m_pRunNumberLabel->setText(QString("<b>%1</b>").arg(runNumber));
341  m_pRunStateLabel->setText(stateToString(m_pRunControlService->getRunState()).c_str());
342  m_pRunStartTimeLabel->setText(startTimeStdStr.c_str());
343  m_pRunEndTimeLabel->setText("NONE");
344  }
345  catch(StatusCodeException &exception)
346  {
347  if(m_pLogger)
348  m_pLogger->log(ERROR, "Couldn't start new run : " + exception.toString());
349  return;
350  }
351 
352  m_pStartOfRunButton->setEnabled(false);
353  m_pEndOfRunButton->setEnabled(true);

425 void DQMRunControlWidget::importFile(const std::string &fileName)
426 {
427  TiXmlDocument xmlDocument(fileName);
428 
429  if (!xmlDocument.LoadFile())
430  {
431  QMessageBox::warning(this, "Warning", "Invalid xml file");
432  return;
433  }
434 
435  const TiXmlHandle xmlDocumentHandle(&xmlDocument);
436  const TiXmlHandle xmlHandle = TiXmlHandle(xmlDocumentHandle.FirstChildElement().Element());
437 
438  TiXmlElement *const pRunInfoElement = xmlHandle.FirstChildElement("runinfo").Element();
439 
440  if(!pRunInfoElement)
441  {
442  QMessageBox::warning(this, "Warning", "Invalid xml file : Couldn't find <runinfo> element !");
443  return;
444  }
445 
446  TiXmlHandle runInfoHandle(pRunInfoElement);
447 
448  int runNumber = 0;
449  std::string detectorName;
450  std::string description;
451 
452  m_pRunParametersTableWidget->clearContents();
453  while (m_pRunParametersTableWidget->rowCount() > 0)
454  m_pRunParametersTableWidget->removeRow(0);
455 
456  try
457  {
458  TiXmlElement *const pRunNumberElement = runInfoHandle.FirstChildElement("runnumber").Element();
459  TiXmlElement *const pDetectorElement = runInfoHandle.FirstChildElement("detector").Element();
460  TiXmlElement *const pDescriptionElement = runInfoHandle.FirstChildElement("description").Element();
461  TiXmlElement *const pParametersElement = runInfoHandle.FirstChildElement("parameters").Element();
462 
463  if(pRunNumberElement)
464  {
465  THROW_RESULT_IF(STATUS_CODE_SUCCESS, !=, DQMXmlHelper::getAttribute(pRunNumberElement, "value", runNumber));
466  }
467 
468  if(pDetectorElement)
469  {
470  THROW_RESULT_IF(STATUS_CODE_SUCCESS, !=, DQMXmlHelper::getAttribute(pDetectorElement, "name", detectorName));
471  }
472 
473  if(pDescriptionElement)
474  {
475  THROW_RESULT_IF(STATUS_CODE_SUCCESS, !=, DQMXmlHelper::getAttribute(pDescriptionElement, "value", description));
476  }
477 
478  if(pParametersElement)
479  {
480  TiXmlHandle parametersHandle(pParametersElement);
481 
482  for (TiXmlElement *pXmlElement = parametersHandle.FirstChild("parameter").Element(); NULL != pXmlElement;
483  pXmlElement = pXmlElement->NextSiblingElement("parameter"))
484  {
485  std::string name;
486  std::string value;
487 
488  THROW_RESULT_IF(STATUS_CODE_SUCCESS, !=, DQMXmlHelper::getAttribute(pXmlElement, "name", name));
489  THROW_RESULT_IF(STATUS_CODE_SUCCESS, !=, DQMXmlHelper::getAttribute(pXmlElement, "value", value));
490 
491  int row = m_pRunParametersTableWidget->rowCount();
492 
493  m_pRunParametersTableWidget->insertRow(row);
494  m_pRunParametersTableWidget->setItem(row, 0, new QTableWidgetItem(name.c_str()));
495  m_pRunParametersTableWidget->setItem(row, 1, new QTableWidgetItem(value.c_str()));
496  }
497  }
498  }
499  catch(const StatusCodeException &exception)
500  {
501  QMessageBox::warning(this, "Warning", ("Invalid xml file : Couldn't parse it : " + exception.toString()).c_str());
502  return;
503  }
504 
505  m_pRunNumberSpinBox->setValue(runNumber);
506  m_pRunDescriptionTextEdit->clear();
507  m_pRunDescriptionTextEdit->setText(description.c_str());
508  m_pDetectorNameEdit->setText(detectorName.c_str());
509 }

521 void DQMRunControlWidget::exportFile(const std::string &fileName)
522 {
523  TiXmlDocument document;
524 
525  TiXmlDeclaration *pDeclaration = new TiXmlDeclaration( "1.0", "", "" );

530 
531  TiXmlElement *pRunInfoElement = new TiXmlElement("runinfo");

533 
534  TiXmlElement *pRunNumberElement = new TiXmlElement("runnumber");

540  pRunInfoElement->LinkEndChild(pDetectorElement);
541 
542  TiXmlElement *pDescriptionElement = new TiXmlElement("description");
543  pDescriptionElement->SetAttribute("value", m_pRunDescriptionTextEdit->toPlainText().toStdString());
544  pRunInfoElement->LinkEndChild(pDescriptionElement);
545 
546 
547  if(m_pRunParametersTableWidget->rowCount() > 0)
548  {
549  TiXmlElement *pParametersElement = new TiXmlElement("parameters");
550  pRunInfoElement->LinkEndChild(pParametersElement);
551 
552  for(unsigned int i=0 ; i<m_pRunParametersTableWidget->rowCount() ; i++)
553  {
554  QTableWidgetItem *pKeyItem = m_pRunParametersTableWidget->item(i, 0);
555  QTableWidgetItem *pValueItem = m_pRunParametersTableWidget->item(i, 1);
556 
557  if(!pKeyItem || !pValueItem)
558  continue;
559 
560  TiXmlElement *pParameterElement = new TiXmlElement("parameter");
561  pParameterElement->SetAttribute("name", pKeyItem->text().toStdString());
562  pParameterElement->SetAttribute("value", pValueItem->text().toStdString());
563  pParametersElement->LinkEndChild(pParameterElement);
564  }
565  }
566
